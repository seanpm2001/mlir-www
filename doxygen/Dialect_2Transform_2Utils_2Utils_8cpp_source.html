28 ParseResult mlir::transform::parsePackedOrDynamicIndexList(
29  OpAsmParser &parser, std::optional<OpAsmParser::UnresolvedOperand> &packed,
30  Type &packedType, SmallVectorImpl<OpAsmParser::UnresolvedOperand> &values,
31  SmallVectorImpl<Type> &valueTypes, DenseI64ArrayAttr &integers) {
32  OpAsmParser::UnresolvedOperand packedOperand;
33  if (parser.parseOptionalStar().succeeded()) {
34  if (parser.parseLParen().failed() ||
35  parser.parseOperand(packedOperand).failed() ||
36  parser.parseColonType(packedType).failed() ||
37  parser.parseRParen().failed()) {
38  return failure();
39  }
40  packed.emplace(packedOperand);
41  integers = parser.getBuilder().getDenseI64ArrayAttr({});
42  return success();
43  }
44 
45  return parseDynamicIndexList(parser, values, integers, &valueTypes);
