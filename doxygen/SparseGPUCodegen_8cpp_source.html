57 /// Constructs a new GPU kernel in the given GPU module.
58 static gpu::GPUFuncOp genGPUFunc(OpBuilder &builder, gpu::GPUModuleOp gpuModule,
59  SmallVectorImpl<Value> &args) {
60  // Get a unique kernel name. Not very creative,
61  // but we simply try kernel0, kernel1, etc.
62  unsigned kernelNumber = 0;
63  SmallString<16> kernelName;
64  do {
65  kernelName.clear();
66  ("kernel" + Twine(kernelNumber++)).toStringRef(kernelName);
67  } while (gpuModule.lookupSymbol(kernelName));
68  // Then we insert a new kernel with given arguments into the module.
69  builder.setInsertionPointToStart(&gpuModule.getBodyRegion().front());
70  SmallVector<Type> argsTp;
71  for (unsigned i = 0, e = args.size(); i < e; i++)
72  argsTp.push_back(args[i].getType());
73  FunctionType type = FunctionType::get(gpuModule->getContext(), argsTp, {});
74  auto gpuFunc =
75  builder.create<gpu::GPUFuncOp>(gpuModule->getLoc(), kernelName, type);
76  gpuFunc->setAttr(gpu::GPUDialect::getKernelFuncAttrName(),
77  builder.getUnitAttr());
78  return gpuFunc;
79 }

99 /// Maps the provided ranked host buffer into the device address space.
100 /// Writes from the host are guaranteed to be visible to device kernels
101 /// that are launched afterwards. Writes from the device are guaranteed
102 /// to be visible on the host after synchronizing with the device kernel
103 /// completion. Needs to cast the buffer to a unranked buffer.
104 static Value genHostRegisterMemref(OpBuilder &builder, Location loc,
105  Value mem) {
106  MemRefType memTp = cast<MemRefType>(mem.getType());
107  UnrankedMemRefType resTp =
108  UnrankedMemRefType::get(memTp.getElementType(), /*memorySpace=*/0);
109  Value cast = builder.create<memref::CastOp>(loc, resTp, mem);
110  builder.create<gpu::HostRegisterOp>(loc, cast);
111  return cast;
112 }

133 /// Allocates memory on the device.
134 /// TODO: A `host_shared` attribute could be used to indicate that
135 /// the buffer is visible by both host and device, but lowering
136 /// that feature does not seem to be fully supported yet.
137 static gpu::AllocOp genAllocMemRef(OpBuilder &builder, Location loc, Value mem,
138  Value token) {
139  auto tp = cast<ShapedType>(mem.getType());
140  auto elemTp = tp.getElementType();
141  auto shape = tp.getShape();
142  auto memTp = MemRefType::get(shape, elemTp);
143  SmallVector<Value> dynamicSizes;
144  for (unsigned r = 0, rank = tp.getRank(); r < rank; r++) {
145  if (shape[r] == ShapedType::kDynamic) {
146  Value dimOp = linalg::createOrFoldDimOp(builder, loc, mem, r);
147  dynamicSizes.push_back(dimOp);
148  }
149  }
150  return builder.create<gpu::AllocOp>(loc, TypeRange({memTp, token.getType()}),
151  token, dynamicSizes, ValueRange());
152 }

177 /// Generates an alloc/copy pair.
178 static Value genAllocCopy(OpBuilder &builder, Location loc, Value b,
179  SmallVectorImpl<Value> &tokens) {
180  Value firstToken = genFirstWait(builder, loc);
181  auto alloc = genAllocMemRef(builder, loc, b, firstToken);
182  Value devMem = alloc.getResult(0);
183  Value depToken = alloc.getAsyncToken(); // copy-after-alloc
184  tokens.push_back(genCopyMemRef(builder, loc, devMem, b, depToken));
185  return devMem;
186 }

197 /// Prepares the outlined arguments, passing scalars and buffers in. Here we
198 /// assume that the first buffer is the one allocated for output. We create
199 /// a set of properly chained asynchronous allocation/copy pairs to increase
200 /// overlap before launching the kernel.
201 /// TODO: the output assumption may be a bit too brittle
202 static Value genParametersIn(OpBuilder &builder, Location loc,
203  SmallVectorImpl<Value> &scalars,
204  SmallVectorImpl<Value> &buffers,
205  SmallVectorImpl<Value> &args,
206  SmallVectorImpl<Value> &tokens,
207  bool useHostRegistrationForOut) {
208  Value out;
209  // Scalars are passed by value.
210  for (Value s : scalars)
211  args.push_back(s);
212  // Buffers are need to be made visible on device.
213  for (Value b : buffers) {
214  if (useHostRegistrationForOut) {
215  out = genHostRegisterMemref(builder, loc, b);
216  args.push_back(b);
217  useHostRegistrationForOut = false;
218  continue;
219  }
220  args.push_back(genAllocCopy(builder, loc, b, tokens));
221  }
222  return out;
223 }

225 /// Finalizes the outlined arguments. The output buffer is copied depending
226 /// on the kernel token and then deallocated. All other buffers are simply
227 /// deallocated. Then we wait for all operations to complete.
228 static void genParametersOut(OpBuilder &builder, Location loc, Value out,
229  Value kernelToken, SmallVectorImpl<Value> &scalars,
230  SmallVectorImpl<Value> &buffers,
231  SmallVectorImpl<Value> &args,
232  SmallVectorImpl<Value> &tokens) {
233  unsigned base = scalars.size();
234  for (unsigned i = base, e = args.size(); i < e; i++) {
235  Value firstToken;
236  if (i == base) {
237  // Assumed output parameter: unregister or copy-out.
238  if (out) {
239  genHostUnregisterMemref(builder, loc, out);
240  out = Value();
241  continue;
242  }
243  firstToken =
244  genCopyMemRef(builder, loc, buffers[0], args[i], kernelToken);
245  } else {
246  firstToken = genFirstWait(builder, loc);
247  }
248  tokens.push_back(genDeallocMemRef(builder, loc, args[i], firstToken));
249  }
250 }

252 /// Constructs code for new GPU kernel.
253 static void genGPUCode(PatternRewriter &rewriter, gpu::GPUFuncOp gpuFunc,
254  scf::ParallelOp forallOp,
255  SmallVectorImpl<Value> &constants,
256  SmallVectorImpl<Value> &scalars,
257  SmallVectorImpl<Value> &buffers) {
258  Location loc = gpuFunc->getLoc();
259  Block &block = gpuFunc.getBody().front();
260  rewriter.setInsertionPointToStart(&block);
261 
262  // Re-generate the constants, recapture all arguments.
263  unsigned arg = 0;
264  IRMapping irMap;
265  for (Value c : constants)
266  irMap.map(c, rewriter.clone(*c.getDefiningOp())->getResult(0));
267  for (Value s : scalars)
268  irMap.map(s, block.getArgument(arg++));
269  for (Value b : buffers)
270  irMap.map(b, block.getArgument(arg++));
271 
272  // Assume 1-dimensional grid/block configuration (only x dimension),
273  // so that:
274  // row = blockIdx.x * blockDim.x + threadIdx.x
275  // inc = blockDim.x * gridDim.x
276  Value bid = rewriter.create<gpu::BlockIdOp>(loc, gpu::Dimension::x);
277  Value bsz = rewriter.create<gpu::BlockDimOp>(loc, gpu::Dimension::x);
278  Value tid = rewriter.create<gpu::ThreadIdOp>(loc, gpu::Dimension::x);
279  Value gsz = rewriter.create<gpu::GridDimOp>(loc, gpu::Dimension::x);
280  Value mul = rewriter.create<arith::MulIOp>(loc, bid, bsz);
281  Value row = rewriter.create<arith::AddIOp>(loc, mul, tid);
282  Value inc = rewriter.create<arith::MulIOp>(loc, bsz, gsz);
283 
284  // Construct the iteration over the computational space that
285  // accounts for the fact that the total number of threads and
286  // the amount of work to be done usually do not match precisely.
287  // for (r = row; r < N; r += inc) {
288  // <loop-body>
289  // }
290  Value upper = irMap.lookup(forallOp.getUpperBound()[0]);
291  scf::ForOp forOp = rewriter.create<scf::ForOp>(loc, row, upper, inc);
292  rewriter.cloneRegionBefore(forallOp.getLoopBody(), forOp.getLoopBody(),
293  forOp.getLoopBody().begin(), irMap);
294 
295  // Done.
296  rewriter.setInsertionPointAfter(forOp);
297  rewriter.create<gpu::ReturnOp>(gpuFunc->getLoc());
298 }

304 /// Helper to detect a * b.
305 static bool matchMulOfArgs(linalg::GenericOp op, Value val) {
306  if (auto *def = val.getDefiningOp()) {
307  if (isa<arith::MulFOp>(def) || isa<arith::MulIOp>(def)) {
308  Value a = op.getBlock()->getArguments()[0];
309  Value b = op.getBlock()->getArguments()[1];
310  return (def->getOperand(0) == a && def->getOperand(1) == b) ||
311  (def->getOperand(0) == b && def->getOperand(1) == a);
312  }
313  }
314  return false;
315 }
316 
317 /// Helper to detect x = x + a * b
318 static bool matchSumOfMultOfArgs(linalg::GenericOp op) {
319  auto yieldOp = cast<linalg::YieldOp>(op.getRegion().front().getTerminator());
320  if (auto *def = yieldOp.getOperand(0).getDefiningOp()) {
321  if (isa<arith::AddFOp>(def) || isa<arith::AddIOp>(def)) {
322  Value x = op.getBlock()->getArguments()[2];
323  return (def->getOperand(0) == x &&
324  matchMulOfArgs(op, def->getOperand(1))) ||
325  (def->getOperand(1) == x &&
326  matchMulOfArgs(op, def->getOperand(0)));
327  }
328  }
329  return false;
330 }

332 /// Test for sorted COO with suitable data and coordinates types.
333 static bool isAdmissibleCOO(SparseTensorType &aTp) {
334  return aTp.isCompressedLvl(0) && aTp.isOrderedLvl(0) && !aTp.isUniqueLvl(0) &&
335  aTp.isSingletonLvl(1) && aTp.isOrderedLvl(1) && aTp.isUniqueLvl(1) &&
336  (aTp.getElementType().isF64() || aTp.getElementType().isF32()) &&
337  (aTp.getCrdWidth() == 0 || aTp.getCrdWidth() == 32 ||
338  aTp.getCrdWidth() == 64);
339 }

341 /// Test for CSR with suitable data and coordinates types.
342 static bool isAdmissibleCSR(SparseTensorType &aTp) {
343  return aTp.isDenseLvl(0) && aTp.isCompressedLvl(1) && aTp.isOrderedLvl(1) &&
344  aTp.isUniqueLvl(1) &&
345  (aTp.getElementType().isF64() || aTp.getElementType().isF32()) &&
346  (aTp.getCrdWidth() == 0 || aTp.getCrdWidth() == 32 ||
347  aTp.getCrdWidth() == 64);
348 }

351 static bool areAdmissibleTypes(SparseTensorType aTp, SparseTensorType bTp,

403 /// Match and rewrite SpMV kernel.
404 static LogicalResult rewriteSpMV(PatternRewriter &rewriter,
405  linalg::GenericOp op, bool enableRT) {
406  Location loc = op.getLoc();
407  Value a = op.getOperand(0);
408  Value x = op.getOperand(1);
409  Value y = op.getOperand(2); // we have y = Ax
410  SmallVector<Value> tokens;
411 
412  // Only admissible sparse matrix format and dense vectors.
413  bool isCOO = false;
414  SparseTensorType aTp = getSparseTensorType(a);
415  SparseTensorType xTp = getSparseTensorType(x);
416  SparseTensorType yTp = getSparseTensorType(y);
417  if (!areAdmissibleTypes(aTp, xTp, yTp, enableRT, isCOO))
418  return failure();
419 
420  // Start sparse kernel and copy data from host to device.
421  // a : memR/memC/memV -> rowA,colA,valA
422  // x : memX -> vecX
423  // y : memY -> vecY
424  Value nseA = rewriter.create<NumberOfEntriesOp>(loc, a);
425  Value szY = linalg::createOrFoldDimOp(rewriter, loc, a, 0);
426  Value szX = linalg::createOrFoldDimOp(rewriter, loc, a, 1);
427  Value memR = genFirstPosOrCrds(rewriter, loc, a, isCOO, enableRT);
428  Value memC = genSecondCrds(rewriter, loc, a, isCOO, enableRT);
429  Value memV = genToValues(rewriter, loc, a);
430  Value rowA = genAllocCopy(rewriter, loc, memR, tokens);
431  Value colA = memC ? genAllocCopy(rewriter, loc, memC, tokens) : Value();
432  Value valA = genAllocCopy(rewriter, loc, memV, tokens);
433  Value memX = genTensorToMemref(rewriter, loc, x);
434  Value vecX = genAllocCopy(rewriter, loc, memX, tokens);
435  Value memY = genTensorToMemref(rewriter, loc, y);
436  Value vecY = genAllocCopy(rewriter, loc, memY, tokens);
437  genBlockingWait(rewriter, loc, tokens);
438  tokens.clear();
439 
440  // Create sparse environment and sparse matrix/dense vector handles.
441  Type indexTp = rewriter.getIndexType();
442  Type envHandleTp = rewriter.getType<gpu::SparseEnvHandleType>();
443  Type dnVecHandleTp = rewriter.getType<gpu::SparseDnVecHandleType>();
444  Type spmatHandleTp = rewriter.getType<gpu::SparseSpMatHandleType>();
445  Type tokenTp = rewriter.getType<gpu::AsyncTokenType>();
446  Value token = genFirstWait(rewriter, loc);
447  auto env =
448  rewriter.create<gpu::CreateSparseEnvOp>(loc, envHandleTp, tokenTp, token);
449  Value handle = env.getResult(0);
450  token = env.getAsyncToken();
451  Operation *spGenA =
452  genSpMat(rewriter, loc, spmatHandleTp, tokenTp, token, szY, szX, nseA,
453  rowA, colA, valA, isCOO, enableRT);
454  Value spMatA = spGenA->getResult(0);
455  token = spGenA->getResult(1);
456  auto dvecX = rewriter.create<gpu::CreateDnVecOp>(loc, dnVecHandleTp, tokenTp,
457  token, vecX, szX);
458  Value dnX = dvecX.getResult(0);
459  token = dvecX.getAsyncToken();
460  auto dvecY = rewriter.create<gpu::CreateDnVecOp>(loc, dnVecHandleTp, tokenTp,
461  token, vecY, szY);
462  Value dnY = dvecY.getResult(0);
463  token = dvecY.getAsyncToken();
464 
465  // Precompute buffersize for SpMV.
466  auto bufferComp = rewriter.create<gpu::SpMVBufferSizeOp>(
467  loc, indexTp, tokenTp, token, handle, spMatA, dnX, dnY);
468  Value bufferSz = bufferComp.getResult(0);
469  token = bufferComp.getAsyncToken();
470  auto buf = genAllocBuffer(rewriter, loc, bufferSz, token);
471  Value buffer = buf.getResult(0);
472  token = buf.getAsyncToken();
473 
474  // Perform the SpMV.
475  auto spmvComp = rewriter.create<gpu::SpMVOp>(loc, tokenTp, token, handle,
476  spMatA, dnX, dnY, buffer);
477  token = spmvComp.getAsyncToken();
478 
479  // Copy data back to host and free all the resoures.
480  token = rewriter.create<gpu::DestroySpMatOp>(loc, tokenTp, token, spMatA)
481  .getAsyncToken();
482  token = rewriter.create<gpu::DestroyDnVecOp>(loc, tokenTp, token, dnX)
483  .getAsyncToken();
484  token = rewriter.create<gpu::DestroyDnVecOp>(loc, tokenTp, token, dnY)
485  .getAsyncToken();
486  token = rewriter.create<gpu::DestroySparseEnvOp>(loc, tokenTp, token, handle)
487  .getAsyncToken();
488  token = genDeallocMemRef(rewriter, loc, rowA, token);
489  if (colA)
490  token = genDeallocMemRef(rewriter, loc, colA, token);
491  token = genDeallocMemRef(rewriter, loc, valA, token);
492  token = genDeallocMemRef(rewriter, loc, buffer, token);
493  token = genDeallocMemRef(rewriter, loc, vecX, token);
494  token = genCopyMemRef(rewriter, loc, memY, vecY, token);
495  token = genDeallocMemRef(rewriter, loc, vecY, token);
496  tokens.push_back(token);
497  genBlockingWait(rewriter, loc, tokens);
498  tokens.clear();
499 
500  // Done.
501  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, memY);
502  return success();
503 }

505 /// Match and rewrite SpMM kernel.
506 static LogicalResult rewriteSpMM(PatternRewriter &rewriter,
507  linalg::GenericOp op, bool enableRT) {
508  Location loc = op.getLoc();
509  Value a = op.getOperand(0);
510  Value b = op.getOperand(1);
511  Value c = op.getOperand(2); // we have C = AB
512  SmallVector<Value> tokens;
513 
514  // Only admissible sparse matrix format and dense matrices.
515  bool isCOO = false;
516  SparseTensorType aTp = getSparseTensorType(a);
517  SparseTensorType bTp = getSparseTensorType(b);
518  SparseTensorType cTp = getSparseTensorType(c);
519  if (!areAdmissibleTypes(aTp, bTp, cTp, enableRT, isCOO))
520  return failure();
521 
522  // Start sparse kernel and copy data from host to device.
523  // a : memR/memC/memV -> rowA,colA,valA
524  // b : bufB -> matA
525  // c : bufC -> matC
526  Value nseA = rewriter.create<NumberOfEntriesOp>(loc, a);
527  Value szm = linalg::createOrFoldDimOp(rewriter, loc, a, 0);
528  Value szk = linalg::createOrFoldDimOp(rewriter, loc, a, 1);
529  Value szn = linalg::createOrFoldDimOp(rewriter, loc, b, 1);
530  Value memR = genFirstPosOrCrds(rewriter, loc, a, isCOO, enableRT);
531  Value memC = genSecondCrds(rewriter, loc, a, isCOO, enableRT);
532  Value memV = genToValues(rewriter, loc, a);
533  Value rowA = genAllocCopy(rewriter, loc, memR, tokens);
534  Value colA = memC ? genAllocCopy(rewriter, loc, memC, tokens) : Value();
535  Value valA = genAllocCopy(rewriter, loc, memV, tokens);
536  Value bufB = genTensorToMemref(rewriter, loc, b);
537  Value matB = genAllocCopy(rewriter, loc, bufB, tokens);
538  Value bufC = genTensorToMemref(rewriter, loc, c);
539  Value matC = genAllocCopy(rewriter, loc, bufC, tokens);
540  genBlockingWait(rewriter, loc, tokens);
541  tokens.clear();
542 
543  // Create sparse environment and sparse matrix/dense matrix handles.
544  Type indexTp = rewriter.getIndexType();
545  Type envHandleTp = rewriter.getType<gpu::SparseEnvHandleType>();
546  Type dnMatHandleTp = rewriter.getType<gpu::SparseDnMatHandleType>();
547  Type spMatHandleTp = rewriter.getType<gpu::SparseSpMatHandleType>();
548  Type tokenTp = rewriter.getType<gpu::AsyncTokenType>();
549  Value token = genFirstWait(rewriter, loc);
550  auto env =
551  rewriter.create<gpu::CreateSparseEnvOp>(loc, envHandleTp, tokenTp, token);
552  Value handle = env.getResult(0);
553  token = env.getAsyncToken();
554  Operation *spGenA =
555  genSpMat(rewriter, loc, spMatHandleTp, tokenTp, token, szm, szk, nseA,
556  rowA, colA, valA, isCOO, enableRT);
557  Value spMatA = spGenA->getResult(0);
558  token = spGenA->getResult(1);
559  auto dmatB = rewriter.create<gpu::CreateDnMatOp>(loc, dnMatHandleTp, tokenTp,
560  token, szk, szn, matB);
561  Value dnB = dmatB.getResult(0);
562  token = dmatB.getAsyncToken();
563  auto dmatC = rewriter.create<gpu::CreateDnMatOp>(loc, dnMatHandleTp, tokenTp,
564  token, szm, szn, matC);
565  Value dnC = dmatC.getResult(0);
566  token = dmatC.getAsyncToken();
567 
568  // Precompute buffersize for SpMM.
569  auto bufferComp = rewriter.create<gpu::SpMMBufferSizeOp>(
570  loc, indexTp, tokenTp, token, handle, spMatA, dnB, dnC);
571  Value bufferSz = bufferComp.getResult(0);
572  token = bufferComp.getAsyncToken();
573  auto buf = genAllocBuffer(rewriter, loc, bufferSz, token);
574  Value buffer = buf.getResult(0);
575  token = buf.getAsyncToken();
576 
577  // Perform the SpMM.
578  auto spmmComp = rewriter.create<gpu::SpMMOp>(loc, tokenTp, token, handle,
579  spMatA, dnB, dnC, buffer);
580  token = spmmComp.getAsyncToken();
581 
582  // Copy data back to host and free all the resoures.
583  token = rewriter.create<gpu::DestroySpMatOp>(loc, tokenTp, token, spMatA)
584  .getAsyncToken();
585  token = rewriter.create<gpu::DestroyDnMatOp>(loc, tokenTp, token, dnB)
586  .getAsyncToken();
587  token = rewriter.create<gpu::DestroyDnMatOp>(loc, tokenTp, token, dnC)
588  .getAsyncToken();
589  token = rewriter.create<gpu::DestroySparseEnvOp>(loc, tokenTp, token, handle)
590  .getAsyncToken();
591  token = genDeallocMemRef(rewriter, loc, rowA, token);
592  if (colA)
593  token = genDeallocMemRef(rewriter, loc, colA, token);
594  token = genDeallocMemRef(rewriter, loc, valA, token);
595  token = genDeallocMemRef(rewriter, loc, buffer, token);
596  token = genDeallocMemRef(rewriter, loc, matB, token);
597  token = genCopyMemRef(rewriter, loc, bufC, matC, token);
598  token = genDeallocMemRef(rewriter, loc, matC, token);
599  tokens.push_back(token);
600  genBlockingWait(rewriter, loc, tokens);
601  tokens.clear();
602 
603  // Done.
604  rewriter.replaceOpWithNewOp<bufferization::ToTensorOp>(op, bufC);
605  return success();
606 }

612 /// Proof-of-concept rewriter. This rule generates a GPU implementation
613 /// for each outermost forall loop generated by the sparse compiler.
614 /// TODO: right works with parallelization-strategy=dense-outer-loop
615 /// but give this its own flags in the future
616 struct ForallRewriter : public OpRewritePattern<scf::ParallelOp> {
617  using OpRewritePattern<scf::ParallelOp>::OpRewritePattern;
618 
619  ForallRewriter(MLIRContext *context, unsigned nT)
620  : OpRewritePattern(context), numThreads(nT){};
621 
622  LogicalResult matchAndRewrite(scf::ParallelOp forallOp,
623  PatternRewriter &rewriter) const override {
624  // Reject inadmissible loop form.
625  // Essentially only accept a loop, generated by the sparse compiler,
626  // of the form
627  // forall (i = 0; i < N; i++)
628  // so that cyclic scheduling over the threads is easy.
629  if (!forallOp->hasAttr(LoopEmitter::getLoopEmitterLoopAttrName()) ||
630  forallOp.getNumReductions() != 0 || forallOp.getNumLoops() != 1 ||
631  !matchPattern(forallOp.getLowerBound()[0], m_Zero()) ||
632  !matchPattern(forallOp.getStep()[0], m_One()))
633  return failure();
634  // Collect every value that is computed outside the parallel loop.
635  SetVector<Value> invariants; // stable iteration!
636  forallOp->walk([&](Operation *op) {
637  // Collect all values of admissible ops.
638  for (OpOperand &o : op->getOpOperands()) {
639  Value val = o.get();
640  Block *block;
641  if (auto arg = dyn_cast<BlockArgument>(val))
642  block = arg.getOwner();
643  else
644  block = val.getDefiningOp()->getBlock();
645  if (!isNestedIn(block, forallOp))
646  invariants.insert(val);
647  }
648  });
649  // Outline the outside values as proper parameters. Fail when sharing
650  // value between host and device is not straightforward.
651  SmallVector<Value> constants;
652  SmallVector<Value> scalars;
653  SmallVector<Value> buffers;
654  for (Value val : invariants) {
655  Type tp = val.getType();
656  if (val.getDefiningOp<arith::ConstantOp>())
657  constants.push_back(val);
658  else if (isa<FloatType>(tp) || tp.isIntOrIndex())
659  scalars.push_back(val);
660  else if (isa<MemRefType>(tp))
661  buffers.push_back(val);
662  else
663  return failure(); // don't know how to share
664  }
665  // Pass outlined non-constant values.
666  // TODO: Experiment with `useHostRegistrationForOut` to see if we want to
667  // keep the feature at all (either through a heuristic or compiler
668  // option for gpu codegen).
669  Location loc = forallOp->getLoc();
670  SmallVector<Value> args;
671  SmallVector<Value> tokens;
672  Value out = genParametersIn(rewriter, loc, scalars, buffers, args, tokens,
673  /*useHostRegistrationForOut=*/false);
674  // Set up GPU module and construct GPU function.
675  auto saveIp = rewriter.saveInsertionPoint();
676  ModuleOp topModule = forallOp->getParentOfType<ModuleOp>();
677  auto gpuModule = genGPUModule(rewriter, topModule);
678  auto gpuFunc = genGPUFunc(rewriter, gpuModule, args);
679  genGPUCode(rewriter, gpuFunc, forallOp, constants, scalars, buffers);
680  // Generate code that launches the kernel asynchronously, blocking on all
681  // opens tokens and yielding a new token for the output.
682  // TODO: Passing in tokens to launch up does not seem to be properly lowered
683  // by cubin yet, hence the current blocking wait.
684  rewriter.restoreInsertionPoint(saveIp);
685  genBlockingWait(rewriter, loc, tokens);
686  tokens.clear();
687  Value kernelToken =
688  genLaunchGPUFunc(rewriter, gpuFunc, args, tokens, numThreads);
689  // Finalize the outlined arguments.
690  genParametersOut(rewriter, loc, out, kernelToken, scalars, buffers, args,
691  tokens);
692  genBlockingWait(rewriter, loc, tokens);
693  rewriter.eraseOp(forallOp);
694  return success();
695  }
696 
697 private:
698  // Helper method to see if block appears in given loop.
699  static bool isNestedIn(Block *block, scf::ParallelOp forallOp) {
700  for (Operation *o = block->getParentOp(); o; o = o->getParentOp()) {
701  if (o == forallOp)
702  return true;
703  }
704  return false;
705  }
706 
707  unsigned numThreads;
708 };

714 /// Proof-of-concept rewriter. This rule recognizes certain math kernels
715 /// and replaces these with corresponding calls into a sparse library.
716 struct LinalgOpRewriter : public OpRewritePattern<linalg::GenericOp> {
717  using OpRewritePattern<linalg::GenericOp>::OpRewritePattern;
718 
719  LinalgOpRewriter(MLIRContext *context, bool rt)
720  : OpRewritePattern(context), enableRT(rt) {}
721 
722  LogicalResult matchAndRewrite(linalg::GenericOp op,
723  PatternRewriter &rewriter) const override {
724  if (op.getNumDpsInits() != 1)
725  return failure(); // reject multi-output
726 
727  const unsigned numLoops = op.getNumLoops();
728  const unsigned numTensors = op->getNumOperands();
729  const auto iteratorTypes = op.getIteratorTypesArray();
730  SmallVector<AffineMap, 4> maps = op.getIndexingMapsArray();
731 
732  using MapList = ArrayRef<ArrayRef<AffineExpr>>;
733  auto infer = [](MapList m) { return AffineMap::inferFromExprList(m); };
734  AffineExpr i, j, k;
735  bindDims(getContext(), i, j, k);
736 
737  // TODO: more robust patterns, tranposed versions, more kernels...
738 
739  // Recognize a SpMV kernel.
740  if (numLoops == 2 && numTensors == 3 &&
741  linalg::isParallelIterator(iteratorTypes[0]) &&
742  linalg::isReductionIterator(iteratorTypes[1]) &&
743  // TODO: add transposed {i, j}
744  maps == infer({{i, j}, {j}, {i}}) && matchSumOfMultOfArgs(op)) {
745  return rewriteSpMV(rewriter, op, enableRT);
746  }
747 
748  // Recognize a SpMM kernel.
749  if (numLoops == 3 && numTensors == 3 &&
750  linalg::isParallelIterator(iteratorTypes[0]) &&
751  linalg::isParallelIterator(iteratorTypes[1]) &&
752  linalg::isReductionIterator(iteratorTypes[2]) &&
753  // TODO: add transposed {i, k}, {k, j}
754  // TODO: maybe add transposed {i, j} in future
755  maps == infer({{i, k}, {k, j}, {i, j}}) && matchSumOfMultOfArgs(op)) {
756  return rewriteSpMM(rewriter, op, enableRT);
757  }
758 
759  return failure();
760  }
761

763  bool enableRT;
764 };
