28 #define CUDA_REPORT_IF_ERROR(expr) \
29  [](CUresult result) { \
30  if (!result) \
31  return; \
32  const char *name = nullptr; \
33  cuGetErrorName(result, &name); \
34  if (!name) \
35  name = "<unknown>"; \
36  fprintf(stderr, "'%s' failed with '%s'\n", #expr, name); \
37  }(expr)
38 
39 #define CUSPARSE_REPORT_IF_ERROR(expr) \
40  { \
41  cusparseStatus_t status = (expr); \
42  if (status != CUSPARSE_STATUS_SUCCESS) { \
43  fprintf(stderr, "cuSPARSE '%s' failed with '%s'\n", #expr, \
44  cusparseGetErrorString(status)); \
45  } \
46  }

50 // Make the primary context of the current default device current for the
51 // duration
52 // of the instance and restore the previous context on destruction.
53 class ScopedContext {
54 public:
55  ScopedContext() {
56  // Static reference to CUDA primary context for device ordinal
57  // defaultDevice.
58  static CUcontext context = [] {
59  CUDA_REPORT_IF_ERROR(cuInit(/*flags=*/0));
60  CUdevice device;
61  CUDA_REPORT_IF_ERROR(cuDeviceGet(&device, /*ordinal=*/defaultDevice));
62  CUcontext ctx;
63  // Note: this does not affect the current context.
64  CUDA_REPORT_IF_ERROR(cuDevicePrimaryCtxRetain(&ctx, device));
65  return ctx;
66  }();
67 
68  CUDA_REPORT_IF_ERROR(cuCtxPushCurrent(context));
69  }
70 
71  ~ScopedContext() { CUDA_REPORT_IF_ERROR(cuCtxPopCurrent(nullptr)); }
72 };

183 /// Registers a memref with the CUDA runtime. `descriptor` is a pointer to a
184 /// ranked memref descriptor struct of rank `rank`. Helpful until we have
185 /// transfer functions implemented.
186 extern "C" MLIR_CUDA_WRAPPERS_EXPORT void
187 mgpuMemHostRegisterMemRef(int64_t rank, StridedMemRefType<char, 1> *descriptor,
188  int64_t elementSizeBytes) {
189  // Only densely packed tensors are currently supported.
190  int64_t *denseStrides = (int64_t *)alloca(rank * sizeof(int64_t));
191  int64_t *sizes = descriptor->sizes;
192  for (int64_t i = rank - 1, runningStride = 1; i >= 0; i--) {
193  denseStrides[i] = runningStride;
194  runningStride *= sizes[i];
195  }
196  uint64_t sizeBytes = sizes[0] * denseStrides[0] * elementSizeBytes;
197  int64_t *strides = &sizes[rank];
198  (void)strides;
199  for (unsigned i = 0; i < rank; ++i)
200  assert(strides[i] == denseStrides[i] &&
201  "Mismatch in computed dense strides");
202 
203  auto *ptr = descriptor->data + descriptor->offset * elementSizeBytes;
204  mgpuMemHostRegister(ptr, sizeBytes);
205 }

249 // Some macro magic to get float/double alpha and beta on host.
250 #define ALPHABETA(w, alpha, beta) \
251  float(alpha##f) = 1.0f; \
252  float(beta##f) = 1.0f; \
253  double(alpha##d) = 1.0; \
254  double(beta##d) = 1.0; \
255  const void *(alpha##p) = nullptr; \
256  const void *(beta##p) = nullptr; \
257  if ((w) == 32) { \
258  (alpha##p) = reinterpret_cast<void *>(&(alpha##f)); \
259  (beta##p) = reinterpret_cast<void *>(&(beta##f)); \
260  } else { \
261  (alpha##p) = reinterpret_cast<void *>(&(alpha##d)); \
262  (beta##p) = reinterpret_cast<void *>(&(beta##d)); \
