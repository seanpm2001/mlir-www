52 /// A helper class around an array of values that corresponds to a sparse
53 /// tensor. This class provides a set of meaningful APIs to query and update
54 /// a particular field in a consistent way. Users should not make assumptions
55 /// on how a sparse tensor is laid out but instead rely on this class to access
56 /// the right value for the right field.
57 template <typename ValueArrayRef>
58 class SparseTensorDescriptorImpl {
59 protected:
60  // TODO: Functions/methods marked with [NUMFIELDS] might should use
61  // `FieldIndex` for their return type, via the same reasoning for why
62  // `Dimension`/`Level` are used both for identifiers and ranks.
63  SparseTensorDescriptorImpl(SparseTensorType stt, ValueArrayRef fields)
64  : rType(stt), fields(fields), layout(stt) {
65  assert(layout.getNumFields() == getNumFields());
66  // We should make sure the class is trivially copyable (and should be small
67  // enough) such that we can pass it by value.
68  static_assert(std::is_trivially_copyable_v<
69  SparseTensorDescriptorImpl<ValueArrayRef>>);
70  }
71 
72 public:
73  FieldIndex getMemRefFieldIndex(SparseTensorFieldKind kind,
74  std::optional<Level> lvl) const {
75  // Delegates to storage layout.
76  return layout.getMemRefFieldIndex(kind, lvl);
77  }
78 
79  // TODO: See note [NUMFIELDS].
80  unsigned getNumFields() const { return fields.size(); }
81 
82  ///
83  /// Getters: get the value for required field.
84  ///
85 
86  Value getSpecifier() const { return fields.back(); }
87 
88  Value getSpecifierField(OpBuilder &builder, Location loc,
89  StorageSpecifierKind kind,
90  std::optional<Level> lvl) const {
91  SparseTensorSpecifier md(fields.back());
92  return md.getSpecifierField(builder, loc, kind, lvl);
93  }
94 
95  Value getLvlSize(OpBuilder &builder, Location loc, Level lvl) const {
96  return getSpecifierField(builder, loc, StorageSpecifierKind::LvlSize, lvl);
97  }
98 
99  Value getPosMemRef(Level lvl) const {
100  return getMemRefField(SparseTensorFieldKind::PosMemRef, lvl);
101  }
102 
103  Value getValMemRef() const {
104  return getMemRefField(SparseTensorFieldKind::ValMemRef, std::nullopt);
105  }
106 
107  Value getMemRefField(SparseTensorFieldKind kind,
108  std::optional<Level> lvl) const {
109  return getField(getMemRefFieldIndex(kind, lvl));
110  }
111 
112  Value getMemRefField(FieldIndex fidx) const {
113  assert(fidx < fields.size() - 1);
114  return getField(fidx);
115  }
116 
117  Value getPosMemSize(OpBuilder &builder, Location loc, Level lvl) const {
118  return getSpecifierField(builder, loc, StorageSpecifierKind::PosMemSize,
119  lvl);
120  }
121 
122  Value getCrdMemSize(OpBuilder &builder, Location loc, Level lvl) const {
123  return getSpecifierField(builder, loc, StorageSpecifierKind::CrdMemSize,
124  lvl);
125  }
126 
127  Value getValMemSize(OpBuilder &builder, Location loc) const {
128  return getSpecifierField(builder, loc, StorageSpecifierKind::ValMemSize,
129  std::nullopt);
130  }
131 
132  Type getMemRefElementType(SparseTensorFieldKind kind,
133  std::optional<Level> lvl) const {
134  return getMemRefType(getMemRefField(kind, lvl)).getElementType();
135  }
136 
137  Value getField(FieldIndex fidx) const {
138  assert(fidx < fields.size());
139  return fields[fidx];
140  }
141 
142  ValueRange getMemRefFields() const {
143  // Drop the last metadata fields.
144  return fields.drop_back();
145  }
146 
147  std::pair<FieldIndex, unsigned> getCrdMemRefIndexAndStride(Level lvl) const {
148  return layout.getFieldIndexAndStride(SparseTensorFieldKind::CrdMemRef, lvl);
149  }
150 
151  Value getAOSMemRef() const {
152  const Level cooStart = getCOOStart(rType.getEncoding());
153  assert(cooStart < rType.getLvlRank());
154  return getMemRefField(SparseTensorFieldKind::CrdMemRef, cooStart);
155  }
156 
157  RankedTensorType getRankedTensorType() const { return rType; }
158  ValueArrayRef getFields() const { return fields; }
159  StorageLayout getLayout() const { return layout; }
160 
161 protected:
162  SparseTensorType rType;
163  ValueArrayRef fields;
164  StorageLayout layout;
165 };

176 /// Uses SmallVectorImpl<Value> & for mutable descriptors.
177 /// Using SmallVector for mutable descriptor allows users to reuse it as a
178 /// tmp buffers to append value for some special cases, though users should
179 /// be responsible to restore the buffer to legal states after their use. It
180 /// is probably not a clean way, but it is the most efficient way to avoid
181 /// copying the fields into another SmallVector. If a more clear way is
182 /// wanted, we should change it to MutableArrayRef instead.
183 class MutSparseTensorDescriptor
184  : public SparseTensorDescriptorImpl<SmallVectorImpl<Value> &> {
185 public:
186  MutSparseTensorDescriptor(SparseTensorType stt,
187  SmallVectorImpl<Value> &buffers)
188  : SparseTensorDescriptorImpl<SmallVectorImpl<Value> &>(stt, buffers) {}
189 
190  // Allow implicit type conversion from mutable descriptors to immutable ones
191  // (but not vice versa).
192  /*implicit*/ operator SparseTensorDescriptor() const {
193  return SparseTensorDescriptor(rType, fields);
194  }

250 inline Value genTuple(OpBuilder &builder, Location loc, Type tp,

261 inline SparseTensorDescriptor getDescriptorFromTensorTuple(Value tensor) {
262  auto tuple = getTuple(tensor);
263  SparseTensorType stt(cast<RankedTensorType>(tuple.getResultTypes()[0]));
264  return SparseTensorDescriptor(stt, tuple.getInputs());
265 }

267 inline MutSparseTensorDescriptor
268 getMutDescriptorFromTensorTuple(Value tensor, SmallVectorImpl<Value> &fields) {
269  auto tuple = getTuple(tensor);
270  fields.assign(tuple.getInputs().begin(), tuple.getInputs().end());
271  SparseTensorType stt(cast<RankedTensorType>(tuple.getResultTypes()[0]));
272  return MutSparseTensorDescriptor(stt, fields);
273 }
