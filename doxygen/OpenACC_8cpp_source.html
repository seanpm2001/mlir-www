270 /// Pattern to remove operation without region that have constant false `ifCond`
271 /// and remove the condition from the operation if the `ifCond` is a true
272 /// constant.
273 template <typename OpTy>
274 struct RemoveConstantIfCondition : public OpRewritePattern<OpTy> {
275  using OpRewritePattern<OpTy>::OpRewritePattern;
276 
277  LogicalResult matchAndRewrite(OpTy op,
278  PatternRewriter &rewriter) const override {
279  // Early return if there is no condition.
280  Value ifCond = op.getIfCond();
281  if (!ifCond)
282  return failure();
283 
284  IntegerAttr constAttr;
285  if (!matchPattern(ifCond, m_Constant(&constAttr)))
286  return failure();
287  if (constAttr.getInt())
288  rewriter.updateRootInPlace(op, [&]() { op.getIfCondMutable().erase(0); });
289  else
290  rewriter.eraseOp(op);
291 
292  return success();
293  }
294 };

296 /// Replaces the given op with the contents of the given single-block region,
297 /// using the operands of the block terminator to replace operation results.
298 static void replaceOpWithRegion(PatternRewriter &rewriter, Operation *op,
299  Region &region, ValueRange blockArgs = {}) {
300  assert(llvm::hasSingleElement(region) && "expected single-region block");
301  Block *block = &region.front();
302  Operation *terminator = block->getTerminator();
303  ValueRange results = terminator->getOperands();
304  rewriter.inlineBlockBefore(block, op, blockArgs);
305  rewriter.replaceOp(op, results);
306  rewriter.eraseOp(terminator);
307 }

309 /// Pattern to remove operation with region that have constant false `ifCond`
310 /// and remove the condition from the operation if the `ifCond` is constant
311 /// true.
312 template <typename OpTy>
313 struct RemoveConstantIfConditionWithRegion : public OpRewritePattern<OpTy> {
314  using OpRewritePattern<OpTy>::OpRewritePattern;
315 
316  LogicalResult matchAndRewrite(OpTy op,
317  PatternRewriter &rewriter) const override {
318  // Early return if there is no condition.
319  Value ifCond = op.getIfCond();
320  if (!ifCond)
321  return failure();
322 
323  IntegerAttr constAttr;
324  if (!matchPattern(ifCond, m_Constant(&constAttr)))
325  return failure();
326  if (constAttr.getInt())
327  rewriter.updateRootInPlace(op, [&]() { op.getIfCondMutable().erase(0); });
328  else
329  replaceOpWithRegion(rewriter, op, op.getRegion());
330 
331  return success();
332  }
333 };

341 static LogicalResult verifyInitLikeSingleArgRegion(
342  Operation *op, Region &region, StringRef regionType, StringRef regionName,
343  Type type, bool verifyYield, bool optional = false) {
344  if (optional && region.empty())
345  return success();
346 
347  if (region.empty())
348  return op->emitOpError() << "expects non-empty " << regionName << " region";
349  Block &firstBlock = region.front();
350  if (firstBlock.getNumArguments() != 1 ||
351  firstBlock.getArgument(0).getType() != type)
352  return op->emitOpError() << "expects " << regionName
353  << " region with one "
354  "argument of the "
355  << regionType << " type";
356 
357  if (verifyYield) {
358  for (YieldOp yieldOp : region.getOps<acc::YieldOp>()) {
359  if (yieldOp.getOperands().size() != 1 ||
360  yieldOp.getOperands().getTypes()[0] != type)
361  return op->emitOpError() << "expects " << regionName
362  << " region to "
363  "yield a value of the "
364  << regionType << " type";
365  }
366  }
367  return success();
368 }

449  llvm::SmallVectorImpl<Type> &types, mlir::ArrayAttr &symbols) {

497 template <typename Op>
498 static LogicalResult
499 checkSymOperandList(Operation *op, std::optional<mlir::ArrayAttr> attributes,
500  mlir::OperandRange operands, llvm::StringRef operandName,
501  llvm::StringRef symbolName) {
502  if (!operands.empty()) {
503  if (!attributes || attributes->size() != operands.size())
504  return op->emitOpError()
505  << "expected as many " << symbolName << " symbol reference as "
506  << operandName << " operands";
507  } else {
508  if (attributes)
509  return op->emitOpError()
510  << "unexpected " << symbolName << " symbol reference";
511  return success();
512  }
513 
514  llvm::DenseSet<Value> set;
515  for (auto args : llvm::zip(operands, *attributes)) {
516  mlir::Value operand = std::get<0>(args);
517 
518  if (!set.insert(operand).second)
519  return op->emitOpError()
520  << operandName << " operand appears more than once";
521 
522  mlir::Type varType = operand.getType();
523  auto symbolRef = llvm::cast<SymbolRefAttr>(std::get<1>(args));
524  auto decl = SymbolTable::lookupNearestSymbolFrom<Op>(op, symbolRef);
525  if (!decl)
526  return op->emitOpError()
527  << "expected symbol reference " << symbolRef << " to point to a "
528  << operandName << " declaration";
529 
530  if (decl.getType() && decl.getType() != varType)
531  return op->emitOpError() << "expected " << operandName << " (" << varType
532  << ") to be the same type as " << operandName
533  << " declaration (" << decl.getType() << ")";
534  }
535 
536  return success();
537 }

637 static ParseResult
638 parseGangClause(OpAsmParser &parser,
639  std::optional<OpAsmParser::UnresolvedOperand> &gangNum,
640  Type &gangNumType,
641  std::optional<OpAsmParser::UnresolvedOperand> &gangStatic,
642  Type &gangStaticType, UnitAttr &hasGang) {
643  hasGang = UnitAttr::get(parser.getBuilder().getContext());
644  // optional gang operands
645  if (succeeded(parser.parseOptionalLParen())) {
646  if (succeeded(parser.parseOptionalKeyword(LoopOp::getGangNumKeyword()))) {
647  if (parser.parseEqual())
648  return failure();
