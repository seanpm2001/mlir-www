17 /// Returns true if all rank reduced in the given `extractOp` happen in leading
18 /// dimensions earlier than last `trailingRank` dimensions.
19 static bool areAllRankReducedLeadingDim(tensor::ExtractSliceOp extractOp,
20  unsigned trailingRank) {
21  // If no ranks are reduced at all, it's a degenerated case; always true.
22  if (extractOp.getSourceType().getRank() == extractOp.getType().getRank())
23  return true;
24 
25  RankedTensorType inferredType = extractOp.inferResultType(
26  extractOp.getSourceType(), extractOp.getMixedOffsets(),
27  extractOp.getMixedSizes(), extractOp.getMixedStrides());
28  return extractOp.getType().getShape().take_back(trailingRank) ==
29  inferredType.getShape().take_back(trailingRank);
30 }

33 /// Fold transfer_reads of a tensor.extract_slice op. E.g.:
34 ///
35 /// ```
36 /// %0 = tensor.extract_slice %t[%a, %b] [%c, %d] [1, 1]
37 /// : tensor<?x?xf32> to tensor<?x?xf32>
38 /// %1 = vector.transfer_read %0[%e, %f], %cst {in_bounds = [true, true]}
39 /// : tensor<?x?xf32>, vector<4x5xf32>
40 /// ```
41 /// is rewritten to:
42 /// ```
43 /// %p0 = arith.addi %a, %e : index
44 /// %p1 = arith.addi %b, %f : index
45 /// %1 = vector.transfer_read %t[%p0, %p1], %cst {in_bounds = [true, true]}
46 /// : tensor<?x?xf32>, vector<4x5xf32>
47 /// ```
48 // TODO: this is brittle and should be deprecated in favor of a more general
49 // pattern that applies on-demand.
50 class FoldExtractSliceIntoTransferRead final
51  : public OpRewritePattern<vector::TransferReadOp> {
52 public:
53  using OpRewritePattern::OpRewritePattern;
54 
55  FoldExtractSliceIntoTransferRead(MLIRContext *context,
56  std::function<bool(Operation *op)> controlFn,
57  PatternBenefit benefit)
58  : OpRewritePattern(context, benefit), controlFn(std::move(controlFn)) {}
59 
60  LogicalResult matchAndRewrite(vector::TransferReadOp xferOp,
61  PatternRewriter &rewriter) const override {
62  if (controlFn && !controlFn(xferOp))
63  return failure();
64 
65  // TODO: support 0-d corner case.
66  if (xferOp.getTransferRank() == 0)
67  return failure();
68  if (xferOp.hasOutOfBoundsDim())
69  return failure();
70  if (!xferOp.getPermutationMap().isMinorIdentity())
71  return failure();
72  if (xferOp.getMask())
73  return failure();
74  auto extractOp = xferOp.getSource().getDefiningOp<tensor::ExtractSliceOp>();
75  if (!extractOp)
76  return failure();
77  if (!extractOp.hasUnitStride())
78  return failure();
79 
80  // Bail on illegal rank-reduction: we need to check that the rank-reduced
81  // dims are exactly the leading dims. I.e. the following is illegal:
82  // ```
83  // %0 = tensor.extract_slice %t[0,0,0][2,1,4][1,1,1] :
84  // tensor<2x1x4xf32> to tensor<2x4xf32>
85  // %1 = vector.transfer_read %0[0,0], %cst :
86  // tensor<2x4xf32>, vector<2x4xf32>
87  // ```
88  //
89  // Cannot fold into:
90  // ```
91  // %0 = vector.transfer_read %t[0,0,0], %cst :
92  // tensor<2x1x4xf32>, vector<2x4xf32>
93  // ```
94  // For this, check the trailing `vectorRank` dims of the extract_slice
95  // result tensor match the trailing dims of the inferred result tensor.
96  if (!areAllRankReducedLeadingDim(extractOp, extractOp.getType().getRank()))
97  return failure();
98 
99  int64_t rankReduced =
100  extractOp.getSourceType().getRank() - extractOp.getType().getRank();
101 
102  SmallVector<Value> newIndices;
103  // In case this is a rank-reducing ExtractSliceOp, copy rank-reduced
104  // indices first.
105  for (int64_t i = 0; i < rankReduced; ++i) {
106  OpFoldResult offset = extractOp.getMixedOffsets()[i];
107  newIndices.push_back(getValueOrCreateConstantIndexOp(
108  rewriter, extractOp.getLoc(), offset));
109  }
110  for (const auto &it : llvm::enumerate(xferOp.getIndices())) {
111  OpFoldResult offset =
112  extractOp.getMixedOffsets()[it.index() + rankReduced];
113  newIndices.push_back(rewriter.create<arith::AddIOp>(
114  xferOp->getLoc(), it.value(),
115  getValueOrCreateConstantIndexOp(rewriter, extractOp.getLoc(),
116  offset)));
117  }
118  SmallVector<bool> inBounds(xferOp.getTransferRank(), true);
119  rewriter.replaceOpWithNewOp<vector::TransferReadOp>(
120  xferOp, xferOp.getVectorType(), extractOp.getSource(), newIndices,
121  xferOp.getPadding(), ArrayRef<bool>{inBounds});
122 
123  return success();
124  }
125 
126 private:
127  std::function<bool(Operation *)> controlFn;
128 };

130 /// Fold tensor.insert_slice into vector.transfer_write if the transfer_write
131 /// could directly write to the insert_slice's destination. E.g.:
132 ///
133 /// ```
134 /// %0 = vector.transfer_write %v, %t1[%c0, %c0] {in_bounds = [true, true]}
135 /// : vector<4x5xf32>, tensor<4x5xf32>
136 /// %1 = tensor.insert_slice %0 into %t2[%a, %b] [4, 5] [1, 1]
137 /// : tensor<4x5xf32> into tensor<?x?xf32>
138 /// ```
139 /// is rewritten to:
140 /// ```
141 /// %1 = vector.transfer_write %v, %t2[%a, %b] {in_bounds = [true, true]}
142 /// : vector<4x5xf32>, tensor<?x?xf32>
143 /// ```
144 // TODO: this is brittle and should be deprecated in favor of a more general
145 // pattern that applies on-demand.
146 class FoldInsertSliceIntoTransferWrite final
147  : public OpRewritePattern<tensor::InsertSliceOp> {
148 public:
149  using OpRewritePattern::OpRewritePattern;
150 
151  FoldInsertSliceIntoTransferWrite(MLIRContext *context,
152  std::function<bool(Operation *op)> controlFn,
153  PatternBenefit benefit)
154  : OpRewritePattern(context, benefit), controlFn(std::move(controlFn)) {}
155 
156  LogicalResult matchAndRewrite(tensor::InsertSliceOp insertOp,
157  PatternRewriter &rewriter) const override {
158  if (!insertOp.hasUnitStride())
159  return failure();
160 
161  auto xferOp = insertOp.getSource().getDefiningOp<vector::TransferWriteOp>();
162  if (!xferOp)
163  return failure();
164  if (controlFn && !controlFn(xferOp))
165  return failure();
166 
167  // TODO: support 0-d corner case.
168  if (xferOp.getTransferRank() == 0)
169  return failure();
170 
171  if (xferOp.hasOutOfBoundsDim())
172  return failure();
173  if (xferOp.getVectorType().getRank() != xferOp.getShapedType().getRank())
174  return failure();
175  if (xferOp.getMask())
176  return failure();
177  // Fold only if the TransferWriteOp completely overwrites the `source` with
178  // a vector. I.e., the result of the TransferWriteOp is a new tensor whose
179  // content is the data of the vector.
180  if (!llvm::equal(xferOp.getVectorType().getShape(),
181  xferOp.getShapedType().getShape()))
182  return failure();
183  if (!xferOp.getPermutationMap().isIdentity())
184  return failure();
185 
186  // Bail on illegal rank-reduction: we need to check that the rank-reduced
187  // dims are exactly the leading dims. I.e. the following is illegal:
188  // ```
189  // %0 = vector.transfer_write %v, %t[0,0], %cst :
190  // vector<2x4xf32>, tensor<2x4xf32>
191  // %1 = tensor.insert_slice %0 into %tt[0,0,0][2,1,4][1,1,1] :
192  // tensor<2x4xf32> into tensor<2x1x4xf32>
193  // ```
194  //
195  // Cannot fold into:
196  // ```
197  // %0 = vector.transfer_write %v, %t[0,0,0], %cst :
198  // vector<2x4xf32>, tensor<2x1x4xf32>
199  // ```
200  // For this, check the trailing `vectorRank` dims of the insert_slice result
201  // tensor match the trailing dims of the inferred result tensor.
202  int64_t rankReduced =
203  insertOp.getType().getRank() - insertOp.getSourceType().getRank();
204  int64_t vectorRank = xferOp.getVectorType().getRank();
205  RankedTensorType inferredSourceTensorType =
206  tensor::ExtractSliceOp::inferResultType(
207  insertOp.getType(), insertOp.getMixedOffsets(),
208  insertOp.getMixedSizes(), insertOp.getMixedStrides());
209  auto actualSourceTensorShape = insertOp.getSourceType().getShape();
210  if (rankReduced > 0 &&
211  actualSourceTensorShape.take_back(vectorRank) !=
212  inferredSourceTensorType.getShape().take_back(vectorRank))
213  return failure();
214 
215  SmallVector<Value> indices = getValueOrCreateConstantIndexOp(
216  rewriter, insertOp.getLoc(), insertOp.getMixedOffsets());
217  SmallVector<bool> inBounds(xferOp.getTransferRank(), true);
218  rewriter.replaceOpWithNewOp<vector::TransferWriteOp>(
219  insertOp, xferOp.getVector(), insertOp.getDest(), indices,
220  ArrayRef<bool>{inBounds});
221  return success();
222  }
223 
224 private:
225  std::function<bool(Operation *)> controlFn;
226 };
