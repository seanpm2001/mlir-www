63 SmallVector<DestructurableMemorySlot> LLVM::AllocaOp::getDestructurableSlots() {
64  if (!mlir::matchPattern(getArraySize(), m_One()))
65  return {};

68  auto destructurable = dyn_cast<DestructurableTypeInterface>(elemType);
69  if (!destructurable)

84 DenseMap<Attribute, MemorySlot>
85 LLVM::AllocaOp::destructure(const DestructurableMemorySlot &slot,
86  const SmallPtrSetImpl<Attribute> &usedIndices,
87  RewriterBase &rewriter) {
88  assert(slot.ptr == getResult());
89  Type elemType =
90  getElemType() ? *getElemType() : getResult().getType().getElementType();
91 
92  rewriter.setInsertionPointAfter(*this);
93 
94  auto destructurableType = cast<DestructurableTypeInterface>(elemType);
95  DenseMap<Attribute, MemorySlot> slotMap;
96  for (Attribute index : usedIndices) {
97  Type elemType = destructurableType.getTypeAtIndex(index);
98  assert(elemType && "used index must exist");
99  auto subAlloca = rewriter.create<LLVM::AllocaOp>(
100  getLoc(), LLVM::LLVMPointerType::get(getContext()), elemType,
101  getArraySize());
102  slotMap.try_emplace<MemorySlot>(index, {subAlloca.getResult(), elemType});
103  }
104 
105  return slotMap;
106 }

130 bool LLVM::LoadOp::canUsesBeRemoved(
131  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
132  SmallVectorImpl<OpOperand *> &newBlockingUses) {
133  if (blockingUses.size() != 1)
134  return false;
135  Value blockingUse = (*blockingUses.begin())->get();
136  // If the blocking use is the slot ptr itself, there will be enough
137  // context to reconstruct the result of the load at removal time, so it can
138  // be removed (provided it loads the exact stored value and is not
139  // volatile).
140  return blockingUse == slot.ptr && getAddr() == slot.ptr &&
141  getResult().getType() == slot.elemType && !getVolatile_();
142 }

153 bool LLVM::StoreOp::canUsesBeRemoved(
154  const MemorySlot &slot, const SmallPtrSetImpl<OpOperand *> &blockingUses,
155  SmallVectorImpl<OpOperand *> &newBlockingUses) {
156  if (blockingUses.size() != 1)
157  return false;
158  Value blockingUse = (*blockingUses.begin())->get();
159  // If the blocking use is the slot ptr itself, dropping the store is
160  // fine, provided we are currently promoting its target value. Don't allow a
161  // store OF the slot pointer, only INTO the slot pointer.
162  return blockingUse == slot.ptr && getAddr() == slot.ptr &&
163  getValue() != slot.ptr && getValue().getType() == slot.elemType &&
164  !getVolatile_();
165 }

283 /// Returns the type the resulting pointer of the GEP points to. If such a type
284 /// is not clear, returns null type.
285 static Type computeReachedGEPType(LLVM::GEPOp gep) {
286  if (gep.getIndices().empty())
287  return {};
288 
289  // Ensures all indices are static and fetches them.
290  SmallVector<IntegerAttr> indices;
291  for (auto index : gep.getIndices()) {
292  IntegerAttr indexInt = llvm::dyn_cast_if_present<IntegerAttr>(index);
293  if (!indexInt)
294  return {};
295  indices.push_back(indexInt);
296  }
297 
298  // Check the pointer indexing only targets the first element.
299  if (indices[0].getInt() != 0)
300  return {};
301 
302  // Set the initial type currently being used for indexing. This will be
303  // updated as the indices get walked over.
304  std::optional<Type> maybeSelectedType = gep.getElemType();
305  if (!maybeSelectedType)
306  return {};
307  Type selectedType = *maybeSelectedType;
308 
309  // Follow the indexed elements in the gep.
310  for (IntegerAttr index : llvm::drop_begin(indices)) {
311  // Ensure the structure of the type being indexed can be reasoned about.
312  // This includes rejecting any potential typed pointer.
313  auto destructurable = llvm::dyn_cast<DestructurableTypeInterface>(selectedType);
314  if (!destructurable)
315  return {};
316 
317  // Follow the type at the index the gep is accessing, making it the new type
318  // used for indexing.
319  Type field = destructurable.getTypeAtIndex(index);
320  if (!field)
321  return {};
322  selectedType = field;
323  }
324 
325  // When there are no more indices, the type currently being used for indexing
326  // is the type of the value pointed at by the returned indexed pointer.
327  return selectedType;
328 }

330 LogicalResult LLVM::GEPOp::ensureOnlySafeAccesses(
331  const MemorySlot &slot, SmallVectorImpl<MemorySlot> &mustBeSafelyUsed) {
332  if (getBase() != slot.ptr)
333  return success();

335  return failure();
336  Type reachedType = computeReachedGEPType(*this);

343 bool LLVM::GEPOp::canRewire(const DestructurableMemorySlot &slot,
344  SmallPtrSetImpl<Attribute> &usedIndices,
345  SmallVectorImpl<MemorySlot> &mustBeSafelyUsed) {
346  auto basePtrType = llvm::dyn_cast<LLVM::LLVMPointerType>(getBase().getType());
347  if (!basePtrType)
348  return false;
349 
350  // Typed pointers are not supported. This should be removed once typed
351  // pointers are removed from the LLVM dialect.
352  if (!basePtrType.isOpaque())
353  return false;
354 
355  if (getBase() != slot.ptr || slot.elemType != getElemType())
356  return false;
357  Type reachedType = computeReachedGEPType(*this);
358  if (!reachedType || getIndices().size() < 2)
359  return false;
360  auto firstLevelIndex = cast<IntegerAttr>(getIndices()[1]);
361  assert(slot.elementPtrs.contains(firstLevelIndex));
362  if (!llvm::isa<LLVM::LLVMPointerType>(slot.elementPtrs.at(firstLevelIndex)))
363  return false;
364  mustBeSafelyUsed.emplace_back<MemorySlot>({getResult(), reachedType});
365  usedIndices.insert(firstLevelIndex);
366  return true;
367 }

369 DeletionKind LLVM::GEPOp::rewire(const DestructurableMemorySlot &slot,
370  DenseMap<Attribute, MemorySlot> &subslots,
371  RewriterBase &rewriter) {
372  IntegerAttr firstLevelIndex = llvm::dyn_cast_if_present<IntegerAttr>(getIndices()[1]);
373  const MemorySlot &newSlot = subslots.at(firstLevelIndex);
374 
375  ArrayRef<int32_t> remainingIndices = getRawConstantIndices().slice(2);
376 
377  // If the GEP would become trivial after this transformation, eliminate it.
378  // A GEP should only be eliminated if it has no indices (except the first
379  // pointer index), as simplifying GEPs with all-zero indices would eliminate
380  // structure information useful for further destruction.
381  if (remainingIndices.empty()) {
382  rewriter.replaceAllUsesWith(getResult(), newSlot.ptr);
383  return DeletionKind::Delete;
384  }
385 
386  rewriter.updateRootInPlace(*this, [&]() {
387  // Rewire the indices by popping off the second index.
388  // Start with a single zero, then add the indices beyond the second.
389  SmallVector<int32_t> newIndices(1);
390  newIndices.append(remainingIndices.begin(), remainingIndices.end());
391  setRawConstantIndices(newIndices);
392 
393  // Rewire the pointed type.
394  setElemType(newSlot.elemType);
395 
396  // Rewire the pointer.
397  getBaseMutable().assign(newSlot.ptr);
398  });
399 
400  return DeletionKind::Keep;
401 }
