101 /// Get the unique ID of a value use. We encode the unique ID combining an owner
102 /// number and the argument number such as if ownerID(op1) < ownerID(op2), then
103 /// useID(op1) < useID(op2). If uses have the same owner, then argNumber(op1) <
104 /// argNumber(op2) implies useID(op1) < useID(op2).
105 template <typename OperandT>
106 static inline uint64_t getUseID(OperandT &val, unsigned ownerID) {
107  uint32_t operandNumberID;
108  if constexpr (std::is_same_v<OpOperand, OperandT>)
109  operandNumberID = val.getOperandNumber();
110  else if constexpr (std::is_same_v<BlockArgument, OperandT>)
111  operandNumberID = val.getArgNumber();
112  else
113  llvm_unreachable("unexpected operand type");
114  return (static_cast<uint64_t>(ownerID) << 32) | operandNumberID;
115 }
