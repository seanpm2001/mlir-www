19 bool isStaticShapeAndContiguousRowMajor(MemRefType type) {
20  if (!type.hasStaticShape())
21  return false;
22 
23  SmallVector<int64_t> strides;
24  int64_t offset;
25  if (failed(getStridesAndOffset(type, strides, offset)))
26  return false;
27 
28  // MemRef is contiguous if outer dimensions are size-1 and inner
29  // dimensions have unit strides.
30  int64_t runningStride = 1;
31  int64_t curDim = strides.size() - 1;
32  // Finds all inner dimensions with unit strides.
33  while (curDim >= 0 && strides[curDim] == runningStride) {
34  runningStride *= type.getDimSize(curDim);
35  --curDim;
36  }
37 
38  // Check if other dimensions are size-1.
39  while (curDim >= 0 && type.getDimSize(curDim) == 1) {
40  --curDim;
41  }
42 
43  // All dims are unit-strided or size-1.
44  return curDim < 0;
45 }
